CREATE TABLE aviso (
  id SERIAL PRIMARY KEY,
  content TEXT NOT NULL,
  aviso_date DATE NOT NULL DEFAULT CURRENT_DATE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE creator (
  id SERIAL PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  hash_code CHAR(8) NOT NULL UNIQUE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE event (
  id SERIAL PRIMARY KEY,
  title VARCHAR(200) NOT NULL,
  description TEXT,
  event_date DATE NOT NULL,
  time_start TIME,
  time_end TIME,
  image_url TEXT,
  type VARCHAR(50) DEFAULT 'Trabalho',
  creator_id INTEGER REFERENCES creator(id),
  responsible_names TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE categoria (
  id SERIAL PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  thumbnail_url TEXT NOT NULL,
  display_order INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE gallery_image (
  id SERIAL PRIMARY KEY,
  categoria_id INTEGER REFERENCES categoria(id) ON DELETE CASCADE,
  image_url TEXT NOT NULL,
  subtitle TEXT,
  display_order INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE schedule_event (
  id SERIAL PRIMARY KEY,
  title VARCHAR(200) NOT NULL,
  subtitle TEXT,
  event_date DATE NOT NULL,
  type VARCHAR(50) DEFAULT 'Prova',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE subject (
  id SERIAL PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  display_order INTEGER DEFAULT 0
);
